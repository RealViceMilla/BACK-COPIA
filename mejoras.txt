----- Modelo Visit -----

Campos que ya tienes:

    nombre
    descripcion
    ubicacion (PointField)
    creado_por (UserProfile)
    fecha_visita

Opcionales / recomendados según funcionalidades extra:

    geofence → PolygonField (área de validación de evidencias)
    fecha_programada → DateTimeField (fecha prevista de la visita, distinta a la creación)
    company → ForeignKey(Company) (asociar visita a empresa)
    activo → BooleanField (habilitar/deshabilitar temporalmente)

----- Modelo Evidence -----

Campos que ya tienes:

    group_member
    visita
    foto
    descripcion
    estado (pendiente, aprobada, rechazada)

Opcionales / recomendados:

    en_geofence → BooleanField (marca automáticamente si la evidencia está dentro del área definida)
    comentarios_estudiante → TextField (observaciones del estudiante)
    ubicacion_foto → PointField (ubicación exacta donde se tomó la evidencia)

----- Modelo Review -----

Campos que ya tienes:

    evidencia (OneToOneField)
    docente (UserProfile)
    validado (BooleanField)
    comentarios
    fecha_revision

Opcionales / recomendados:

    fecha_revision_programada → DateTimeField (para planificar revisiones futuras)
    tipo_revision → CharField (ej.: parcial, final, interna, externa)

----- Modelos UserProfile / Group / Company -----

Opcionales / recomendados:

    telefono → CharField (contacto de usuario, docente o empresa)
    activo → BooleanField (usuario o grupo activo/inactivo)
    empresa_contacto → CharField (empresa asociada a la visita si no se usa ForeignKey)

    from django.contrib.gis.db import models as gis_models
    from django.contrib.auth.models import User
    from django.db import models

    # ----- Modelo de Gestión de Usuarios y Roles -----
    class UserProfile(models.Model):
        ROLE_CHOICES = [
            ('estudiante', 'Estudiante'),
            ('docente', 'Docente Responsable'),
            ('apoyo', 'Docente Apoyo TI'),
            ('coordinador', 'Coordinador API'),
            ('fg', 'Formación General'),
            ('admin', 'Administrador'),
        ]
        user = models.OneToOneField(User, on_delete=models.CASCADE, related_name='profile')
        role = models.CharField(max_length=20, choices=ROLE_CHOICES, default='estudiante')
        telefono = models.CharField(max_length=20, blank=True, null=True)  # Opcional
        activo = models.BooleanField(default=True)  # Usuario activo/inactivo

        def __str__(self):
            full_name = self.user.get_full_name()
            return f"{full_name if full_name else self.user.username} - {self.get_role_display()}"


    # ----- Modelos de Estructura Académica -----
    class Year(models.Model):
        nombre = models.CharField(max_length=20, unique=True)  # Ej: "2025", "2025-2026"
        fecha_inicio = models.DateField(blank=True, null=True)
        fecha_fin = models.DateField(blank=True, null=True)

        def __str__(self):
            return self.nombre


    class Semester(models.Model):
        year = models.ForeignKey(Year, on_delete=models.CASCADE, related_name='semesters', null=True)
        nombre = models.CharField(max_length=50)  # Ej: "Otoño" o "Primavera"

        def __str__(self):
            return f"{self.nombre} {self.year.nombre}"


    class Career(models.Model):
        nombre = models.CharField(max_length=200)
        codigo = models.CharField(max_length=20, unique=True)
        semester = models.ForeignKey(Semester, on_delete=models.CASCADE, related_name='careers')

        def __str__(self):
            return f"{self.nombre} - {self.semester}"


    class Course(models.Model):
        nombre = models.CharField(max_length=200)
        career = models.ForeignKey(Career, on_delete=models.CASCADE, related_name='courses')

        def __str__(self):
            return f"{self.nombre} - {self.career.nombre}"


    class Section(models.Model):
        course = models.ForeignKey(Course, on_delete=models.CASCADE, related_name='sections')
        nombre = models.CharField(max_length=50)  # Ej: "Sección A"
        docentes = models.ManyToManyField(UserProfile, related_name="sections")  # Relación con docentes

        def __str__(self):
            return f"{self.course.nombre} - {self.nombre}"


    class Group(models.Model):
        section = models.ForeignKey(Section, on_delete=models.CASCADE, related_name='groups')
        nombre = models.CharField(max_length=100)
        activo = models.BooleanField(default=True)  # Grupo activo/inactivo

        def __str__(self):
            return f"{self.section} - {self.nombre}"


    class GroupMember(models.Model):
        group = models.ForeignKey(Group, on_delete=models.CASCADE, related_name='members')
        user = models.ForeignKey(User, on_delete=models.CASCADE, related_name='group_memberships')

        class Meta:
            unique_together = ('group', 'user')

        def __str__(self):
            return f"{self.group} :: {self.user.username}"


    # ----- Modelo de Empresas -----
    class Company(models.Model):
        nombre = models.CharField(max_length=200)
        direccion = models.CharField(max_length=250, blank=True)
        ubicacion = gis_models.PointField(help_text='Ubicación principal (lat/lng) de la empresa', srid=4326)
        telefono = models.CharField(max_length=20, blank=True, null=True)  # Opcional

        def __str__(self):
            return self.nombre


    # ----- Modelo de Visitas -----
    class Visit(models.Model):
        creado_por = models.ForeignKey(UserProfile, on_delete=models.CASCADE)
        nombre = models.CharField(max_length=200)
        descripcion = models.TextField()
        ubicacion = gis_models.PointField(srid=4326)
        geofence = gis_models.PolygonField(srid=4326, null=True, blank=True)  # Área de validación
        company = models.ForeignKey(Company, on_delete=models.SET_NULL, null=True, blank=True)  # Empresa asociada
        fecha_visita = models.DateTimeField(auto_now_add=True)
        fecha_programada = models.DateTimeField(blank=True, null=True)  # Fecha prevista de la visita
        activo = models.BooleanField(default=True)  # Habilitar/deshabilitar temporalmente

        def __str__(self):
            return f'Visita de {self.creado_por.user.username} el {self.fecha_visita.date()}'


    # ----- Modelo de Evidencias de Estudiantes -----
    class Evidence(models.Model):
        ESTADOS_EVIDENCIA = [
            ('pendiente', 'Pendiente'),
            ('aprobada', 'Aprobada'),
            ('rechazada', 'Rechazada'),
        ]
        group_member = models.ForeignKey(GroupMember, on_delete=models.CASCADE)
        visita = models.ForeignKey(Visit, on_delete=models.CASCADE)
        foto = models.ImageField(upload_to='evidencias/')
        descripcion = models.TextField(blank=True, null=True)
        comentarios_estudiante = models.TextField(blank=True, null=True)  # Observaciones adicionales
        ubicacion_foto = gis_models.PointField(srid=4326, blank=True, null=True)
        tomado_en = models.DateTimeField(auto_now_add=True)
        estado = models.CharField(max_length=20, choices=ESTADOS_EVIDENCIA, default='pendiente')
        en_geofence = models.BooleanField(default=False)  # Marca si la evidencia está dentro del área

        def actualizar_estado(self):
            try:
                review = self.review
                self.estado = 'aprobada' if review.validado else 'rechazada'
            except Review.DoesNotExist:
                self.estado = 'pendiente'
            self.save()

        def __str__(self):
            return f'Evidencia de la visita {self.visita.id}'


    # ----- Modelo de Revisiones de Docentes -----
    class Review(models.Model):
        evidencia = models.OneToOneField(Evidence, on_delete=models.CASCADE)
        docente = models.ForeignKey(UserProfile, on_delete=models.CASCADE)
        validado = models.BooleanField(default=False)
        comentarios = models.TextField(blank=True, null=True)
        fecha_revision = models.DateTimeField(auto_now_add=True)
        fecha_revision_programada = models.DateTimeField(blank=True, null=True)  # Revisión futura
        tipo_revision = models.CharField(max_length=50, blank=True, null=True)  # Ej: parcial, final, interna, externa

        def save(self, *args, **kwargs):
            super().save(*args, **kwargs)
            self.evidencia.actualizar_estado()

        def __str__(self):
            return f'Revisión de la evidencia {self.evidencia.id}'
